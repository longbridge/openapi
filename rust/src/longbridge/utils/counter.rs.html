1use std::{collections::HashSet, sync::OnceLock};

3static US_ETF_SET: OnceLock<HashSet<&'static str>> = OnceLock::new();
4
5fn us_etf_set() -> &'static HashSet<&'static str> {
6    US_ETF_SET.get_or_init(|| {
7        include_str!("US-ETF.csv")
8            .lines()
9            .map(str::trim)
10            .filter(|s| !s.is_empty())
11            .collect()
12    })
13}

15/// Convert a symbol (e.g. `TSLA.US`) to a counter_id (e.g. `ST/US/TSLA`).
16/// US ETFs are detected via embedded list.
17pub(crate) fn symbol_to_counter_id(symbol: &str) -> String {
18    if let Some((code, market)) = symbol.rsplit_once('.') {
19        let market = market.to_uppercase();
20        let etf_candidate = format!("ETF/{market}/{code}");
21        if us_etf_set().contains(etf_candidate.as_str()) {
22            etf_candidate
23        } else {
24            format!("ST/{market}/{code}")
25        }
26    } else {
27        symbol.to_string()
28    }
29}

31/// Convert an index symbol (e.g. `HSI.HK`) to counter_id (e.g. `IX/HK/HSI`).
32pub(crate) fn index_symbol_to_counter_id(symbol: &str) -> String {
33    if let Some((code, market)) = symbol.rsplit_once('.') {
34        format!("IX/{}/{code}", market.to_uppercase())
35    } else {
36        symbol.to_string()
37    }
38}

40/// Convert a counter_id (e.g. `ST/US/TSLA`) back to a symbol (e.g. `TSLA.US`).
41pub(crate) fn counter_id_to_symbol(counter_id: &str) -> String {
42    let parts: Vec<&str> = counter_id.splitn(3, '/').collect();
43    if parts.len() == 3 {
44        format!("{}.{}", parts[2], parts[1])
45    } else {
46        counter_id.to_string()
47    }
48}

50/// serde deserializer: reads a `counter_id` string and converts it to a symbol.
51pub(crate) fn deserialize_counter_id_as_symbol<'de, D>(d: D) -> Result<String, D::Error>
52where
53    D: serde::Deserializer<'de>,
54{
55    use serde::Deserialize;
56    let counter_id = String::deserialize(d)?;
57    Ok(counter_id_to_symbol(&counter_id))
58}
